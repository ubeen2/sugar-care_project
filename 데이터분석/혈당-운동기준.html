positive_words = ["좋다", "효과", "내려갔다", "정상", "개선", "권장", "꾸준히"]
negative_words = ["힘들다", "부족", "올랐다", "못하겠다"]

def sentiment_score(text):
    pos = sum(word in text for word in positive_words)
    neg = sum(word in text for word in negative_words)
    if pos > neg:
        return "긍정"
    elif neg > pos:
        return "부정"
    else:
        return "중립"

df["감성"] = df["내용"].apply(sentiment_score)

import pandas as pd


docs = [
    "걷기 꾸준히 해서 혈당 내려갔다",
    "HIIT 했는데 너무 힘들었다",
    "자전거 타니까 효과 있었다",
    "스트레칭만 해도 안정되는 느낌",
    "스프린트 했는데 혈당은 내려갔지만 너무 버거웠다"
]


intensity_dict = {
    "저강도": ["걷기", "산책", "스트레칭", "요가"],
    "중강도": ["자전거", "조깅", "수영", "빠르게 걷기"],
    "고강도": ["HIIT", "스프린트", "크로스핏", "고강도"]
}


positive_words = ["내려갔다", "좋았다", "효과", "안정", "정상"]
negative_words = ["힘들었다", "버거웠다", "못하겠다", "오히려 올랐다"]

def classify_intensity(text):
    for level, keywords in intensity_dict.items():
        for k in keywords:
            if k in text:
                return level
    return "미분류"

def classify_sentiment(text):
    for w in positive_words:
        if w in text:
            return "긍정"
    for w in negative_words:
        if w in text:
            return "부정"
    return "중립"


rows = []
for t in docs:
    rows.append({
        "원문": t,
        "강도": classify_intensity(t),
        "감성": classify_sentiment(t)
    })

df = pd.DataFrame(rows)


print(df.to_string(index=False))

import requests
import json
import pandas as pd


client_id = "b7XDFmkGyijVFZNWXjf_"
client_secret = "RYZa30kuAG"

def naver_search(query, display=10, start=1, sort="sim", target="blog"):
    """
    query  : 검색어 (예: '혈당 낮추는 운동')
    display: 한 번에 가져올 개수 (최대 100)
    start  : 검색 시작 위치
    sort   : sim (유사도순), date (날짜순)
    target : 'blog' 또는 'cafearticle'
    """
    url = f"https://openapi.naver.com/v1/search/{target}.json"
    headers = {"X-Naver-Client-Id": client_id,
               "X-Naver-Client-Secret": client_secret}
    params = {"query": query, "display": display, "start": start, "sort": sort}
    res = requests.get(url, headers=headers, params=params)
    
    if res.status_code == 200:
        return res.json()
    else:
        print("❌ Error:", res.status_code)
        return None


queries = ["혈당 낮추는 운동", "공복 혈당 운동", "고강도 운동 혈당", "중강도 운동 혈당"]

all_results = []

for q in queries:
    data_blog = naver_search(q, display=30, target="blog")  # 블로그
    data_cafe = naver_search(q, display=30, target="cafearticle")  # 카페
    
    for d in [data_blog, data_cafe]:
        if d and "items" in d:
            for item in d["items"]:
                all_results.append({
                    "검색어": q,
                    "제목": item["title"].replace("<b>","").replace("</b>",""),
                    "요약": item["description"].replace("<b>","").replace("</b>",""),
                    "링크": item["link"]
                })


df = pd.DataFrame(all_results)
print(df.head(20))

def naver_search(query, display=30, target="blog"):
    """
    네이버 검색 API 호출
    query: 검색어
    display: 결과 개수
    target: blog / cafearticle
    """
    url = f"https://openapi.naver.com/v1/search/{target}.json"
    headers = {"X-Naver-Client-Id": client_id,

    params = {"query": query, "display": display}

    return res.json()

def convert_blog_url(url):
    """네이버 블로그 URL을 본문 iframe용 URL로 변환"""
    m = re.search(r"blog.naver.com/([^/]+)/(\d+)", url)
    if m:
        blogId, logNo = m.groups()
        return f"https://blog.naver.com/PostView.naver?blogId={blogId}&logNo={logNo}"
    return url

def fetch_blog_text(url):
    try:
        url = convert_blog_url(url)
        res = requests.get(url, headers={"User-Agent": "Mozilla/5.0"})
        if res.status_code != 200:
            return ""
        soup = BeautifulSoup(res.text, "html.parser")

        # ✅ iframe 내부 URL 찾기
        iframe = soup.select_one("iframe#mainFrame")
        if iframe and iframe.get("src"):
            iframe_url = "https://blog.naver.com" + iframe["src"]
            res_iframe = requests.get(iframe_url, headers={"User-Agent": "Mozilla/5.0"})
            soup_iframe = BeautifulSoup(res_iframe.text, "html.parser")
            text = soup_iframe.get_text(" ", strip=True)
        else:
            text = soup.get_text(" ", strip=True)

        return text
    except Exception as e:
        return str(e)


exercise_keywords = [
    "걷기", "달리기", "자전거", "수영", "등산", "요가", "필라테스",
    "헬스", "웨이트", "근력", "스트레칭", "유산소", "무산소", "HIIT", "인터벌"
]

def extract_exercises(text):
    return [kw for kw in exercise_keywords if kw in text]

            "링크": row["링크"]
        })

results_df = pd.DataFrame(results)

expanded_rows = []
for _, row in results_df.iterrows():
    if isinstance(row["운동"], str):
