def load_xml_to_df(file):
    tree = ET.parse(file)
    root = tree.getroot()

    rows = []
    for event in root.find("glucose_level").findall("event"):
        ts = event.attrib["ts"]
        value = event.attrib["value"]
        rows.append([ts, float(value)])

    return pd.DataFrame(rows, columns=["timestamp", "glucose"])

def make_features(df, horizon=24):
    """
    df: 한 환자의 DataFrame (index=timestamp, column=glucose)
    horizon: 예측할 미래 스텝 (5분 단위 → 120분 = 24스텝)
    """
    df = df.reset_index(drop=True)
    rows = []
    
    for i in range(len(df) - horizon):
        current = df.loc[i, "glucose"]
        delta = df.loc[i, "glucose"] - df.loc[i-1, "glucose"] if i > 0 else 0
        target = df.loc[i + horizon, "glucose"]  # 120분 후 혈당
        
        rows.append([current, delta, target])
    
    return pd.DataFrame(rows, columns=["current_glucose", "delta", "target"])

def make_features(df, horizon=24, step=5):
    """
    df: 한 환자 DataFrame (index=timestamp, column=glucose)
    horizon: 예측할 미래 스텝 (기본 24 → 120분 뒤, 5분 단위)
    step: 리샘플링 간격 (분 단위)
    """
    df = df.reset_index(drop=True)
    rows = []
    
    for i in range(len(df) - horizon):
        current = df.loc[i, "glucose"]
        
        
        delta = df.loc[i, "glucose"] - df.loc[i-1, "glucose"] if i > 0 else 0
        
        
        mean30 = df.loc[max(0, i-6):i, "glucose"].mean()
        mean60 = df.loc[max(0, i-12):i, "glucose"].mean()
        std30  = df.loc[max(0, i-6):i, "glucose"].std()
        
        
        
        
        target = df.loc[i + horizon, "glucose"]
        
        rows.append([current, delta, mean30, mean60, std30, target])
    
    return pd.DataFrame(rows, columns=["current","delta","mean30","mean60","std30","target"])

from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, mean_squared_error
import pandas as pd

results = []

for i in range(3):
    train = pd.concat([features[j] for j in range(3) if j != i])
    test = features[i]

    
    def normalize(df):
        mean = df["current"].mean()
        std = df["current"].std()
        df_norm = df.copy()
        df_norm[["current","delta","mean30","mean60","std30","target"]] = (
            df_norm[["current","delta","mean30","mean60","std30","target"]] - mean
        ) / std
        return df_norm, mean, std

    train_norm, train_mean, train_std = normalize(train)
    test_norm, test_mean, test_std = normalize(test)

    
    train_norm = train_norm.fillna(0)
    test_norm = test_norm.fillna(0)

    
    X_train, y_train = train_norm.drop(columns=["target"]), train_norm["target"]
    X_test, y_test = test_norm.drop(columns=["target"]), test_norm["target"]

    
    model = RandomForestRegressor(random_state=42, n_estimators=200, max_depth=10)
    model.fit(X_train, y_train)

    
    preds_norm = model.predict(X_test)
    preds = preds_norm * test_std + test_mean
    y_test_real = y_test * test_std + test_mean

    mae = mean_absolute_error(y_test_real, preds)
    rmse = mean_squared_error(y_test_real, preds, squared=False)
    results.append({"환자": f"환자 {i+1}", "MAE": mae, "RMSE": rmse})

    print(f"환자 {i+1} 테스트 결과 → MAE: {mae:.2f}, RMSE: {rmse:.2f}")

results_df = pd.DataFrame(results)
print("\n=== LOPO (정규화+결측처리 버전) 결과 ===")
print(results_df)
